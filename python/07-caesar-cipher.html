In cryptography, a Caesar cipher is one of the simplest encryptopon techniques. For example, with a right shift of 3, A would be replaced by D. Julius Caesar used this method in in his private correspondence.

#!/usr/bin/python3
# 07-caesar-cipher.py

def caesar(type, text, key):
    translated = ""
    text = text.lower()

    if type == "d" or type == "D" or type == "decrypt" or type == "Decrypt":
        key = -key

    for symbol in text:
        if symbol.isalpha():
            num = ord(symbol)
            num += key

            if num > ord('z'):
                num -= 26
            elif num < ord('a'):
                num += 26

            translated += chr(num)

        else:
            translated += symbol

    return "Message: " + translated

print("Welcome to Caesar Cipher")
while True:
    type = input("Encrypt or Decrypt: ")
    text = input("Text: ")
    key = int(input("Key: "))
    print(caesar(type, text, key))
