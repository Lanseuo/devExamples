#!/usr/bin/python3
# 02-hello-world.py

TODO = []

def start():
...global a_b
...showTasks()
...print("What do you want to do?")
...print("(A) Create Task")
...print("(B) Check Task")
...a_b = input("A or B: ")
...if a_b == "a" or a_b == "A":
......createTask()
...elif a_b == "b" or a_b == "B":
......checkTask()
...else:
......print("Choose A or B!")
......pass

def showTasks():
...global TODO
...if TODO:
......print("Your exercises:")
......for i in TODO:
.........print("- ", end=" ")
.........print(i)
...else:
......print("Congratulations! You have nothing to do")

def createTask():
...global TODO
...print("Which task do you want to add?")
...task = input("Task: ")
...TODO.append(task)

def checkTask():
...global TODO
...print("Which task do you want to select?")
...number = int(input("Number: "))
...print("This one: ", end="")
...print(TODO[number - 1], end="")
...print("?")
...really = input("Y/N: ")
...if really == "Y" or really == "y":
......del TODO[number - 1]

print("Welcome to your To-Do-List")
start()

while True:
...showTasks()
...a_b = input("Create Task (A) or Check Task (B): ")
...if a_b == "a" or a_b == "A":
......createTask()
...elif a_b == "b" or a_b == "B":
......checkTask()
